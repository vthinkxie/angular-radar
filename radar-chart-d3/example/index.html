var mycfg = {
  color: function(){
  c = ['red', 'yellow', 'pink', 'green', 'blue', 'olive', 'aqua', 'cadetblue', 'crimson'];
  m = c.length - 1;
  x = parseInt(Math.random()*100);
  return c[x%m]; //Get a random color
}
}
RadarChart.draw("#chart3", d, mycfg);
